# Calculate the 99% confidence interval for the sample average of 'price'
sample_size <- sum(!is.na(pset3dta$price))
sample_mean <- mean(pset3dta$price, na.rm = TRUE)
sample_sd <- sd(pset3dta$price, na.rm = TRUE)

# Z value for 99% confidence level
z_value <- qnorm(0.995) # since it is 2-tailed

margin_of_error <- z_value * (sample_sd / sqrt(sample_size))

ci_lower <- sample_mean - margin_of_error
ci_upper <- sample_mean + margin_of_error

# Print the confidence interval
cat("99% Confidence Interval for the average price is: [", ci_lower, ",", ci_upper, "]\n")
## 99% Confidence Interval for the average price is: [ 17.8578 , 23.68806 ]

This is due to the Frisch-Waugh-Lovell theorem, which states that the coefficients obtained from a regression after partialling out the effects of other variables are the same as those obtained from a multiple regression including all variables.

Q7. five steps in Hypothesis Testing
# Step1 Null Hypothesis: The marginal effect of lprice on lqu is -4,  Alternative Hypothesis: he marginal effect of lprice on lqu is not -4 (greater than or less than)
# 0.05 sig level
t_stat2 <- (beta_hat-(-4))/se_beta_hat
print(t_stat2)
## [1] 1.163376
p_value <- 2 * pt(-abs(t_stat2), df)

# Output the test statistic and p-value
cat("Test Statistic: ", t_stat2, "\nP-Value: ", p_value, "\n")
## Test Statistic:  1.163376 
## P-Value:  0.2496103
#tc=2.00

# Decision rule
if(p_value < 0.05) {
  cat("Reject the null hypothesis: the effect is significantly different from -4.\n")
} else {
  cat("Fail to reject the null hypothesis: insufficient evidence to conclude the effect is different from -4.\n")
}
## Fail to reject the null hypothesis: insufficient evidence to conclude the effect is different from -4.
